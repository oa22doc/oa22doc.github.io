Summary of API Changes

Following is a summary of the API changes between the OpenAccess p028 release and the p021 release.

#################################
# The following classes were modified #
#################################
 



	LayerMapIn 


		Added API

		add(oaUInt2 strmLayer, oaUInt2 strmType, oaLayerNum layer, oaPurposeNum purpose, oaMaterial materialName=oacOtherMaterial, oaUInt4 maskNum=oacNullIndex) 

		add(oaUInt2 strmLayer, oaUInt2 strmType, const oaString &layerName, const oaString &purposeName, oaMaterial materialName=oacOtherMaterial, oaUInt4 maskNum=oacNullIndex)


		getMaterial(oaUInt4 index) const 

		setLpp(oaUInt4 index, oaLayerNum layer, oaPurposeNum purpose)


		Removed API

		add(oaUInt2 strmLayer, oaUInt2 strmType, oaLayerNum layer, oaPurposeNum purpose, oaString *layerName=NULL, oaString *purposeName=NULL, oaString *materialName=NULL, oaUInt4 maskNum=oacNullIndex)

		findLppForStrmLayerTypePair(oaUInt2 strmLayer, oaUInt2 strmType, oaLayerNum &layer, oaPurposeNum &purpose) const


		getMaterialName(oaUInt4 index) const

		isLayerAndPurPoseNotDefined(oaUInt4 index)

		setOaLayer(oaUInt4 index, oaLayerNum layer)

		setOaPurpose(oaUInt4 index, oaPurposeNum purpose)

		
	OptionsIn 


		Added API


		getTextHeight() const 

		setTextHeight(oaUInt4 value)



	StrmIn 


	Added API

		close(Structure *structIn, oaDesign *design) 

		getCustomViaDefStat() const

		getStdViaDefStat() const 

		getTmpCellName() const 

		getTmpViewName() const 

		incrCustomViaDefStat() 

		incrStdViaDefStat() 

		setTmpCellName(oaScalarName tmpCellNameIn) 

		setTmpViewName(oaScalarName tmpViewNameIn) 

		writeStdVia(oaDesign *cell, oaStdViaDef *def, oaPoint &origin, oaBoolean mx, oaDouble angle, oaDouble mag, const PropArray &props, const oaViaParam *params=NULL) 


		Removed API 

		close(oaDesign *design)

		writeStdVia(oaDesign *cell, oaStdViaDef *def, oaPoint &origin, oaBoolean mx, oaDouble angle, oaDouble mag, const PropArray &props)

	Structure 


		Added API

		getBBox() const

		getSavedDesign() const

		getViaParam() const

		setBBox(const oaBox &bBoxIn)

		setSavedDesign(oaBoolean value)

		setViaParam(const oaViaParam *viaParamIn)

#################################
