Summary of API Changes

Following is a summary of the API changes between the OpenAccess p048 source code and documentation release and the p045 source code and documentation release.

#################################
# The following new classes are added #
#################################
 
	oaAnalysisPointArray 
  	oaBuildInfoArray 
	oaMemoryError


#################################


#################################
# The following classes were modified #
#################################

	oaBuildInfo   
		Added API
  		getPackages(oaBuildInfoArray &packagesIn) 


  	oaParasiticNetwork
		Added API
  		create(oaDesignObject *net, const oaAnalysisPointArray &aps) 
	oaTech
		Added API
  		getClearanceMeasure(oaBoolean local) const 
  		getDBUPerUU(const oaViewType *viewType, oaBoolean local) const 
  		getDefaultClearanceMeasure() 
  		getDefaultDefaultManufacturingGrid() 
  		getDefaultIsGateGrounded() 
  		getDefaultManufacturingGrid(oaBoolean local) const 
  		getUserUnits(const oaViewType *viewType, oaBoolean local) const 
  		isClearanceMeasureSet() 
  		isDBUPerUUSet(const oaViewType *viewType) 
  		isDefaultManufacturingGridSet() 
  		isGateGrounded(oaBoolean local) const 
  		isGateGroundedSet() 
  		isUserUnitsSet(const oaViewType *viewType) 
  		setDefaultManufacturingGrid(oaDist grid) 
  		unsetClearanceMeasure() 
  		unsetDBUPerUU(const oaViewType *viewType) 
  		unsetDefaultManufacturingGrid() 
  		unsetGateGrounded() 
  		unsetUserUnits(const oaViewType *viewType)


#################################
