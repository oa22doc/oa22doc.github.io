Summary of API Changes

Following is a summary of the API changes between the OpenAccess p054 release and the p049 release.

#################################
# The following enums were modified #
#################################

	LefDefMsgIds 


		Added Values

		cPhysOnlyInstTermNetMustBeInSameOcc 

		cMultipleInfluenceMaxRange


#################################

#################################
# The following classes were modified #
#################################

	LefOutLayer 


		Added API

		outputEnclosure(oaConstraint *c, oaUInt4 length, oaUInt4 extraCut, oaString side, oaBoolean hard, oaBoolean nonWidth=false)

#################################

#################################
# The following enums have changed values #
#################################

	DirectionEnum 


		Changed Values

		cDirectionRight     New Value:  0 Last Value:  6

		cDirectionUpRight     New Value:  1 Last Value:  7

		cDirectionUp     New Value:  2 Last Value:  0

		cDirectionUpLeft     New Value:  3 Last Value:  1

		cDirectionLeft     New Value:  4 Last Value:  2

		cDirectionDownLeft     New Value:  5 Last Value:  3

		cDirectionDown     New Value:  6 Last Value:  4

		cDirectionDownRight     New Value:  7 Last Value:  5

	LefDefMsgIds 


		Changed Values

		cMacroIllegalSize     New Value:  50145 Last Value:  50146

		cDifferentCumAntennaValue     New Value:  50146 Last Value:  50145


#################################

		setEnclosureConstraint(oaBoolean hard, oaUInt4 index, oaBoolean nonWidth=false)

	LefOutLayer 


		Added API

		writeEnclosure(oaString side, oaBoolean hard) 


		Removed API

		writeEnclosure(oaBoolean hard)
