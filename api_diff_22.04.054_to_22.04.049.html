Summary of API Changes

Following is a summary of the API changes between the OpenAccess p054 release and the p049 release.

#################################
# The following classes were modified #
#################################

	oaArrayInst 


		Added API

		copy(const oaScalarName &name, const oaTransform &xform, oaBlock *block) 

		copy(const oaScalarName &name, const oaTransform &xform) 

	oaDesign 


		Added API

		saveAs(const oaScalarName &libNameIn, const oaScalarName &cellNameIn, const oaScalarName &viewNameIn, oaBoolean renameTopModule) 

	oaScalarInst 


		Added API

		copy(const oaScalarName &name, const oaTransform &xform, oaBlock *block) 

		copy(const oaScalarName &name, const oaTransform &xform) 

	oaVectorInst 


		Added API

		copy(const oaScalarName &baseName, const oaTransform &xform, oaBlock *block) 

		copy(const oaScalarName &baseName, const oaTransform &xform) 

	oaVectorInstBit 


		Added API

		copy(const oaScalarName &baseName, const oaTransform &xform, oaBlock *block) 

		copy(const oaScalarName &baseName, const oaTransform &xform)
