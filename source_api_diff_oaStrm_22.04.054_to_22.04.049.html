Summary of API Changes

Following is a summary of the API changes between the OpenAccess p054 release and the p049 release.

#################################
# The following classes were modified #
#################################

	CellMapOut 


		Added API

		isTranslated(oaUInt4 index) const 

		setTranslated(oaUInt4 index) 

	LayerMapIn 


		Added API

		getNumUsed() const 

		read(const oaString &fileName) 

	PropMapIn 


		Added API

		add(oaUInt2 strmId, oaUInt2 objType, const oaString &propName) 

		read(const oaString &fileName) 


		Removed API

		add(oaUInt2 strmId, oaUInt2 objType, oaString &propName)

	PropMapOut 


		Added API

		add(const oaString &propName, oaUInt2 objType, oaUInt2 strmId) 


		Removed API

		add(oaString &propName, oaUInt2 objType, oaUInt2 strmId)

		isValidLPP(const oaBox &box, oaUInt2 strmLayer, oaUInt2 strmType) const 

		writeOaStdViaCutStruct(oaViaParam &params, oaDesign *design, oaBoolean floating) 

		writeStructBegin(const oaString &name, oaDesign *design) 


		Removed API

		writeStructBegin(const oaString &name)

#################################
# The following classes were modified #
#################################

	CellMapOut 


		Added API

		add(const oaString &libName, const oaString &cellName, const oaString &viewName, const oaString &structName, oaBoolean floating=false) 

		find(const oaString &libName, const oaString &cellName, const oaString &viewName, oaBoolean floating=false) const 


		Removed API

		add(const oaString &libName, const oaString &cellName, const oaString &viewName, const oaString &structName)

		find(const oaString &libName, const oaString &cellName, const oaString &viewName) const

	CellMapOutHashTbl 


		Added API

		find(const oaString &libName, const oaString &cellName, const oaString &viewName, oaBoolean floating) const 

	LayerMapIn 


		Removed API

		findLpp(oaLayerNum layer, oaPurposeNum purpose) const

	LayerMapOut 


		Added API

		add(oaLayerNum layer, oaPurposeNum purpose, oaUInt2 strmLayer, oaUInt2 strmType, oaBoolean floating=false) 

		find(oaLayerNum layer, oaPurposeNum purpose, oaBoolean floating=false) const 

		find(oaLayerNum layer, oaPurposeNum purpose, StrmLayerNumArray &strmLayers, StrmTypeArray &strmTypes, oaBoolean floating=false) const 

		remove(oaLayerNum layer, oaPurposeNum purpose, oaBoolean floating=false) const 


		Removed API

		add(oaLayerNum layer, oaPurposeNum purpose, oaUInt2 strmLayer, oaUInt2 strmType)

		find(oaLayerNum layer, oaPurposeNum purpose, StrmLayerNumArray &strmLayers, StrmTypeArray &strmTypes) const

		find(oaLayerNum layer, oaPurposeNum purpose) const

		remove(oaLayerNum layer, oaPurposeNum purpose)

	LayerMapOutHashTbl 


		Added API

		find(oaLayerNum layer, oaPurposeNum purpose, oaBoolean floating=false) const 

	StrmOut 


		Added API

		getCellName(const oaString &libName, const oaString &cellName, const oaString &viewName, oaString &name, oaBoolean floating) 

		isCustomViaDefMaster(oaCell *cell) 

		mapLayer(oaLayerNum layer, oaPurposeNum purpose, StrmLayerNumArray &strmLayer, StrmTypeArray &strmType, oaBoolean floating=false, oaBoolean exclusive=false) 

		translate(const oaScalarName &lib, const oaScalarName &cell, const oaScalarName &view, oaInt4 depth, oaBoolean force=false, StrmFloatingEnum floating=strmcUnspecified) 

		translateDesign(oaDesign *design, const oaString &structName, oaUInt4 depth, StrmFloatingEnum floating=strmcUnspecified) 

		writeOaDesign(oaDesign *design, const oaString &name, const oaBoolean labelPins=false, StrmFloatingEnum floating=strmcUnspecified) 

		writeOaLPPHeader(oaLPPHeader *header, StrmFloatingEnum floating=strmcUnspecified) 

		writeOaPcell(oaDesign *design, oaBoolean floating) 

		writeOaStdViaCutAref(oaViaParam &params, oaBoolean floating) 

		writeOaStdViaCutStruct(oaViaParam &params, oaBoolean floating) 


		Removed API

		getCellName(const oaString &libName, const oaString &cellName, const oaString &viewName, oaString &name)

		getCellName(const oaDesign *design, oaString &name)

		mapLayer(oaLayerNum layer, oaPurposeNum purpose, StrmLayerNumArray &strmLayer, StrmTypeArray &strmType)

		translate(const oaScalarName &lib, const oaScalarName &cell, const oaScalarName &view, oaInt4 depth, oaBoolean force=false)

		translateDesign(oaDesign *design, const oaString &structName, oaUInt4 depth)

		writeOaDesign(oaDesign *design, const oaString &name, const oaBoolean labelPins=false)

		writeOaLPPHeader(oaLPPHeader *header)

		writeOaPcell(oaDesign *design)

		writeOaStdViaCutAref(oaViaParam &params)

		writeOaStdViaCutStruct(oaViaParam &params)
