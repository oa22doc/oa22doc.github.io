00003 // oaNet.h
00004 //
00005 // This file contains the definition for the oaNet and oaBitNet classes and
00006 // their public supporting classes. The oaNet class is an abstract base for
00007 // all types of nets. The oaBitNet class is the base class for all single-bit
00008 // nets. The supporting classes include:
00009 //
00010 //  oaNetAttrType
00011 //      This class implements an enumerated type that designates different
00012 //      net attributes for use with the oaAttributeDisplay object.
00013 //
00014 //  oaRoutePattern
00015 //      This class implements an enumerated type that describes the routing
00016 //      pattern for the associated net.
00017 //
00018 //  oaConnStatus
00019 //      This class implements an enumerated type that describes the connection
00020 //      status for the associated net.
00021 //
00022 //  oaMemNetIter
00023 //      This class implements an iterator over the collection of nets that
00024 //      a specified net is a member of.
