00003 // oaFigGroup.h
00004 //
00005 // This file contains the definition for the oaFigGroup class and its public
00006 // supporting classes. oaFigGroup is a container object used to encapsulate a
00007 // number of oaFigs such that they can be treated as a single oaFig object.
00008 // An oaFigGroup is itself an oaFig, so oaFigGroups can be nested within one
00009 // another.
00010 //
00011 // The public supporting class included here is:
00012 //
00013 //  oaFigGroupMem
00014 //      Each fig is encapsulated in an oaFigGroupMem for membership in an
00015 //      oaFigGroup.
00016 //
00017 //  oaFigGroupStatus
00018 //      The oaFigGroupStatus type describes the values that the status of an
00019 //      oaFigGroup can take.
